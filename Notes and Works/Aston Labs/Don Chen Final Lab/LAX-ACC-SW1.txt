! initial setup
enable
conf t
enable secret cisco
username cisco secret cisco
line con 0
logging synchronous
login local
exit
line vty 0 4
logging synchronous
login local
exit
hostname LAX-ACC-SW1
no ip routing
end


! setting up some access ports
conf t
int range fa0/4
desc TO_END_HOST
switchport mode access
switchport access vlan 405
shut
no shut
end


! setting up some access ports
conf t
int range fa0/3
desc TO_END_HOST
switchport mode access
switchport access vlan 101
shut
no shut
end


! enables portfast and BPDU guard on an interface
conf t
int fa0/3
	spanning-tree portfast
	spanning-tree bpduguard enable
	end
	
! enables portfast and BPDU guard on an interface
conf t
int f0/4
	spanning-tree portfast
	spanning-tree bpduguard enable
	end



! creating vlans
conf t
vlan 101
name MANAGEMENT_VLAN
vlan 405
name USERS_VLAN
end


! Creates SVI
conf t
int vlan 101
desc MANAGEMENT_VLAN_INTERFACE
ip address 172.16.0.5 255.255.255.240
shut
no shut
end


! setting up a trunk
conf t
int fa0/1
desc TO_LAX-RTR1
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allowed vlan 101,405
end

! setting up a trunk
conf t
int fa0/2
desc TO_LAX-RTR2
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allowed vlan 101,405
end
